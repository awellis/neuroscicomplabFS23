Aggregierte Kennzahlen

se <- function(x) {
  sd(x) / sqrt(length(x))
}
dprimes <- d |>
  select(group, dprime) |> 
  group_by(group) |> 
  summarize(mean = mean(dprime),
            se = se(dprime))
cs <- d |> 
  select(group, c) |> 
  group_by(group) |> 
  summarize(mean = mean(c),
            se = se(c))
